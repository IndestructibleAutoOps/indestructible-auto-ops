name: GL Governance Validation

on:
  push:
    branches: [ main, develop ]
    paths:
      - 'engine/**'
      - 'file-organizer-system/**'
      - 'instant/**'
      - '.github/governance/**'
      - '.github/agents/**'
  pull_request:
    branches: [ main, develop ]
    paths:
      - 'engine/**'
      - 'file-organizer-system/**'
      - 'instant/**'
      - '.github/governance/**'
      - '.github/agents/**'
  workflow_dispatch:

  schedule:
    - cron: '0 0 * * *' # Daily audit at midnight UTC

jobs:
  gov-engine-validation:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      
      - name: Install GL Engine Dependencies
        run: |
          cd engine
          npm ci
          
      - name: Run GL Semantic Validation - Engine
        run: |
          cd engine
          npx ts-node governance/gov_engine.ts validate --strict
          
      - name: Generate Governance Event Stream
        if: failure()
        run: |
          cd engine
          npx ts-node governance/events_writer.ts --event-type=validation_failure
          
      - name: Upload Governance Artifacts
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: gov-governance-engine
          path: |
            engine/.governance/event-stream.jsonl
            engine/.governance/evidence-chain.jsonl

  gov-file-organizer-validation:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      
      - name: Install GL Engine Dependencies
        run: |
          cd engine
          npm ci
          
      - name: Run GL Semantic Validation - File Organizer
        run: |
          npx ts-node engine/governance/gov_engine.ts validate --strict --workspace=file-organizer-system
          
      - name: Generate Governance Event Stream
        if: failure()
        run: |
          npx ts-node engine/governance/events_writer.ts --event-type=validation_failure
          
      - name: Upload Governance Artifacts
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: gov-governance-file-organizer
          path: |
            file-organizer-system/.governance/event-stream.jsonl
            file-organizer-system/.governance/evidence-chain.jsonl

  gov-instant-validation:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      
      - name: Install GL Engine Dependencies
        run: |
          cd engine
          npm ci
          
      - name: Run GL Semantic Validation - Instant
        run: |
          npx ts-node engine/governance/gov_engine.ts validate --strict --workspace=instant
          
      - name: Generate Governance Event Stream
        if: failure()
        run: |
          npx ts-node engine/governance/events_writer.ts --event-type=validation_failure
          
      - name: Upload Governance Artifacts
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: gov-governance-instant
          path: |
            instant/.governance/event-stream.jsonl
            instant/.governance/evidence-chain.jsonl

  gov-gate-execution:
    runs-on: ubuntu-latest
    needs: [gov-engine-validation, gov-file-organizer-validation, gov-instant-validation]
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      
      - name: Install GL Engine Dependencies
        run: |
          cd engine
          npm ci
          
      - name: Execute All GL Gates
        run: |
          cd engine
          npx ts-node gov-gate/core/GateExecutor.ts executeAll
          
      - name: Generate Governance Summary
        if: always()
        run: |
          cd engine
          npx ts-node gov-gate/gates/GovernanceSummaryGate.ts
          
      - name: Upload Gate Reports
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: gov-gate-reports
          path: |
            engine/.governance/gate-reports/*.json

  gov-audit-history-sync:
    runs-on: ubuntu-latest
    needs: [gov-gate-execution]
    if: github.event_name == 'schedule'
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      
      - name: Generate Audit History Report
        run: |
          cd engine
          npx ts-node governance/gov_engine.ts audit-history --days=7
          
      - name: Sync Audit Results
        run: |
          mkdir -p engine/.governance/audit-results/history
          cp engine/.governance/audit-reports/GLOBAL_GOVERNANCE_AUDIT_REPORT.json engine/.governance/audit-results/history/audit-$(date +%Y%m%d).json
          
      - name: Upload Audit History
        uses: actions/upload-artifact@v3
        with:
          name: gov-audit-history
          path: |
            engine/.governance/audit-results/history/
