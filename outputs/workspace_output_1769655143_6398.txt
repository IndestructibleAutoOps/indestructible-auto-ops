export interface FabricEdge {
  id: string;
  sourceId: string;
  targetId: string;
  type: EdgeType;
  layer: FabricLayer;
  
  properties: Record<string, any>;
  superposition: SuperpositionState;
  
  weight: number;
  direction: 'directed' | 'undirected' | 'bidirectional';
  
  version: string;
  realityId: string;
  timestamp: number;
}

export type EdgeType = 
  | 'dependency'     // 依賴關係
  | 'semantic'       // 語意關係
