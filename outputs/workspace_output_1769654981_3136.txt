      case 'normalize':
        return { normalized: true, data: input, preserveOrder: params.preserveOrder };
      
      case 'encode':
        return { encoded: true, data: input };
      
      case 'decode':
        return { decoded: true, data: input };
      
      case 'compress':
        return { compressed: true, data: input, compressed: params.compress };
      
      default:
        return input;
    }
